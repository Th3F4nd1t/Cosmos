hostname COSMOS-RED
!
! VLAN Definitions
vlan 1
    name MANAGEMENT
vlan 10
    name RED_1
vlan 20
    name RED_2
vlan 30
    name RED_3
vlan 100
    name FMS
vlan 200
    name VENUE_UPLINK
!
! Access ports for DS
interface gi1/1
    description RED1 ds port
    switchport mode access
    switchport access vlan 10
    no shutdown
!
interface gi1/2
    description RED2 ds port
    switchport mode access
    switchport access vlan 20
    no shutdown
!
interface gi1/3
    description RED3 ds port
    switchport mode access
    switchport access vlan 30
    no shutdown
!
! PLC
interface gi1/4
    description PLC port
    switchport mode access
    switchport access vlan 100
    no shutdown
!
! unused
interface gi1/5-7
    description UNUSED
    shutdown
!
! Uplink
interface gi1/8
    description Uplink port
    switchport mode trunk
    switchport trunk allowed vlan 1, 10, 20, 30, 100
    no shutdown
! 
! Management interface (static IP)
interface vlan 100
    ip address 10.0.100.150 255.255.255.0
    no shutdown
!
! Default Gateway
ip default-gateway 10.0.100.200
!
! SSH setup
ip domain-name cosmos-red.local
crypto key generate rsa modulus 2048
ip ssh version 2
!
username admin privilege 15 secret <ADMIN_PASSWORD>
!
line vty 0 4
    transport input ssh
    login local
!
end
