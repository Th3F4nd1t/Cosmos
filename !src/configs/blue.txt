hostname COSMOS-BLUE
!
! VLAN Definitions
vlan 1
    name MANAGEMENT
vlan 40
    name BLUE_1
vlan 50
    name BLUE_2
vlan 60
    name BLUE_3
vlan 100
    name FMS
vlan 200
    name VENUE_UPLINK
!
! Access ports for DS
interface gi1/1
    description BLUE1 ds port
    switchport mode access
    switchport access vlan 40
    no shutdown
!
interface gi1/2
    description BLUE2 ds port
    switchport mode access
    switchport access vlan 50
    no shutdown
!
interface gi1/3
    description BLUE3 ds port
    switchport mode access
    switchport access vlan 60
    no shutdown
!
! PLC
interface gi1/4
    description PLC port
    switchport mode access
    switchport access vlan 100
    no shutdown
!
! unused
interface gi1/5-7
    description UNUSED
    shutdown
!
! Uplink
interface gi1/8
    description Uplink port
    switchport mode trunk
    switchport trunk allowed vlan 1, 40, 50, 60, 100
    no shutdown
! 
! Management interface (static IP)
interface vlan 100
    ip address 10.0.100.151 255.255.255.0
    no shutdown
!
! Default Gateway
ip default-gateway 10.0.100.200
!
! SSH setup
ip domain-name cosmos-blue.local
crypto key generate rsa modulus 2048
ip ssh version 2
!
username admin privilege 15 secret <ADMIN_PASSWORD>
!
line vty 0 4
    transport input ssh
    login local
!
end
